IGNORE: "[ \n\r\t]+|;[^\n]*\n";
PREFIX: "`|'|,@|,";

SYMBOL: "[^\"\.`',; \n\r\t\(\)\[\]]+";
STRING: "\"([^\"]|\\\")*\"";

valuelist: value valuelist | value;
value: <prefixed> | <SYMBOL> | <STRING> | <nil> | ["("] <cell> [")"] | ["["] <cell> ["]"];
prefixed: PREFIX value;
cell: value cell | value ["."] value | value;
nil: ["(" ")"] | "nil";
